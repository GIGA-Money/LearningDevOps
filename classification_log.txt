def process(x):
   print("Entering process")
   if x > 5:
      print("x is big")
      if x > 10:
         print("x is very big!")
   elif x < 0:
      print("x is negative")
   print("Exiting process")

def factorial(x):
   if x == 0: 
      return 1
   else:   
      return x * factorial(x-1)
